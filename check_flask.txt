Run these commands:

# 1. Check if Flask is listening on port 5010
sudo netstat -tlnp | grep 5010

# 2. Try to connect directly to Flask
curl -v http://localhost:5010 2>&1

# 3. Check the latest app logs to see if Flask started
tail -30 /var/www/shirzadBot/logs/app.log | grep -i flask

# 4. Check Nginx error log
sudo tail -10 /var/log/nginx/error.log

# 5. Test Nginx can reach localhost:5010
curl -v http://127.0.0.1:5010 2>&1
